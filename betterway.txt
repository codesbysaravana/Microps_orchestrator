// 1. Define your connection variables (store these in your .env)
const JENKINS_URL = 'http://<YOUR_EC2_PUBLIC_IP>:8080';
const JOB_NAME = 'Microps-Core-Pipeline'; 
const JENKINS_USER = 'admin'; // Your Jenkins username
const JENKINS_TOKEN = '11a2b3c4d5e6f7g8h9i0j...'; // The API Token you just generated

// 2. Construct the specific endpoint
const triggerUrl = `${JENKINS_URL}/job/${JOB_NAME}/buildWithParameters`;

// 3. Define the parameters your Jenkinsfile expects
const jobParams = new URLSearchParams({
    TENANT_ID: 'tenant-992',
    REPO_URL: 'https://github.com/someuser/their-app.git',
    BRANCH: 'main'
});

async function triggerJenkinsBuild() {
    try {
        const response = await fetch(`${triggerUrl}?${jobParams.toString()}`, {
            method: 'POST',
            headers: {
                // Encode credentials into a Basic Auth header
                'Authorization': 'Basic ' + Buffer.from(`${JENKINS_USER}:${JENKINS_TOKEN}`).toString('base64')
            }
        });

        // Jenkins returns a 201 Created status code when a build is successfully queued
        if (response.status === 201) {
            console.log('✅ Build successfully queued in Jenkins!');
            
            // Jenkins returns a queue location in the headers, useful for tracking status later
            const queueUrl = response.headers.get('location');
            console.log('Queue URL:', queueUrl);
        } else {
            console.error('❌ Failed to trigger build. Status:', response.status);
            const text = await response.text();
            console.error('Response:', text);
        }
    } catch (error) {
        console.error('🚨 Network error reaching Jenkins:', error);
    }
}

triggerJenkinsBuild();
