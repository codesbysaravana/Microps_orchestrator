1. Dockerize (DockerFile)
2. docker has 2 steps 1. build step (dependency install and run) and 2. run step (run the container)

steps to build and run locally
(do) ---> docker build -t microps-server . (build the image) 
(do) ---> docker run -p 5000:5000 microps-server (run the container)

3. log into the docker into ecr with aws id
(do) ---> aws ecr get-login-password --region ap-southeast-2 | docker login --username AWS --password-stdin 688567265418.dkr.ecr.ap-southeast-2.amazonaws.com
4. tag the docker image in ecr 
(do) ---> docker tag microps-server:latest 688567265418.dkr.ecr.ap-southeast-2.amazonaws.com/microps-server:latest
5. push the docker image to ecr 
(do) ---> docker push 688567265418.dkr.ecr.ap-southeast-2.amazonaws.com/microps-server:latest

// now aws ecr is live after these steps ---> 688567265418.dkr.ecr.ap-southeast-2.amazonaws.com/microps-server:latest


now register from ECR to ECS
-------------------------------------------------> register as the code from deployECS 
aws ecs register-task-definition \
  --family microps-orchestrator-backend \
  --network-mode awsvpc \
  --requires-compatibilities FARGATE \
  --cpu "256" \
  --memory "512" \
  --execution-role-arn "arn:aws:iam::688567265418:role/microps-ecs-execution-role" \
  --task-role-arn "arn:aws:iam::688567265418:role/microps-ecs-task-role" \
  --container-definitions '[{
    "name": "microps-server",
    "image": "688567265418.dkr.ecr.ap-southeast-2.amazonaws.com/microps-server:latest",
    "essential": true,
    "portMappings": [{ "containerPort": 5000, "protocol": "tcp" }],
    "logConfiguration": {
      "logDriver": "awslogs",
      "options": {
        "awslogs-group": "/ecs/microps-backend",
        "awslogs-region": "ap-southeast-2",
        "awslogs-stream-prefix": "server",
        "awslogs-create-group": "true"
      }
    }
  }]' \
  --region ap-southeast-2
